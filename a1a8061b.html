配列の要素追加

コード例

TypeScript

let a = [];

a.push(3);
console.log(a); // [3]

a.push(10, 20);
console.log(a); // [3, 10, 20]

// ...で配列の中身を展開する。
a.push(...[100, 200], 999);
console.log(a); // [3, 10, 20, 100, 200, 999]

// concatは呼び出し元を直接変更するわけではない。
a = a.concat([-10, -20]);
console.log(a); // [3, 10, 20, 100, 200, 999, -10, -20]
