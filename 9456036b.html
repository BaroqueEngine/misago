uniform vec2 uResolution;
varying vec2 vPosition;

void main() {
  vPosition = aPosition / uResolution;

}

varying vec2 vPosition;

void main() {
  gl_FragColor = vec4(vPosition.x, 0.0, 0.0, 1.0);
